import 'package:dio/dio.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';

import '../../../../core/constants/api_constants.dart';
import '../../../../core/network/api_exceptions.dart';
import '../../../../core/network/dio_client.dart';
import '../../../../core/network/dio_provider.dart';
import '../models/my_invitation_model.dart';
import '../models/owner_invitation_model.dart';
import '../models/public_invitation_model.dart';

class InvitationDatasource {
  final DioClient _client;
  const InvitationDatasource(this._client);

  Future<PublicInvitationModel> lookupByToken(String token) async {
    try {
      final res = await _client.dio.get(ApiConstants.invitationLookup(token));
      return PublicInvitationModel.fromJson(
        (res.data as Map<String, dynamic>)['data'] as Map<String, dynamic>,
      );
    } on DioException catch (e) {
      throw mapDioException(e);
    }
  }

  Future<List<MyInvitationModel>> listMine() async {
    try {
      final res = await _client.dio.get(ApiConstants.myInvitations);
      final data = (res.data as Map<String, dynamic>)['data'] as List<dynamic>;
      return data
          .map((e) => MyInvitationModel.fromJson(e as Map<String, dynamic>))
          .toList();
    } on DioException catch (e) {
      throw mapDioException(e);
    }
  }

  Future<void> accept(int id) async {
    try {
      await _client.dio.post(ApiConstants.acceptInvitation(id));
    } on DioException catch (e) {
      throw mapDioException(e);
    }
  }

  Future<void> refuse(int id) async {
    try {
      await _client.dio.post(ApiConstants.refuseInvitation(id));
    } on DioException catch (e) {
      throw mapDioException(e);
    }
  }

  Future<OwnerInvitationModel> sendInvitation({
    required String email,
    String? firstName,
    String? lastName,
    List<String>? specialties,
  }) async {
    try {
      final res = await _client.dio.post(
        ApiConstants.myCompanyEmployeesInvite,
        data: {
          'email': email,
          if (firstName != null && firstName.trim().isNotEmpty) 'first_name': firstName.trim(),
          if (lastName != null && lastName.trim().isNotEmpty) 'last_name': lastName.trim(),
          if (specialties != null && specialties.isNotEmpty) 'specialties': specialties,
        },
      );
      return OwnerInvitationModel.fromJson(
        (res.data as Map<String, dynamic>)['data'] as Map<String, dynamic>,
      );
    } on DioException catch (e) {
      throw mapDioException(e);
    }
  }

  Future<void> resendInvitation(int id) async {
    try {
      await _client.dio.post(ApiConstants.myCompanyInvitationResend(id));
    } on DioException catch (e) {
      throw mapDioException(e);
    }
  }

  Future<void> revokeInvitation(int id) async {
    try {
      await _client.dio.delete(ApiConstants.myCompanyInvitationRevoke(id));
    } on DioException catch (e) {
      throw mapDioException(e);
    }
  }
}

final invitationDatasourceProvider = Provider<InvitationDatasource>((ref) {
  return InvitationDatasource(ref.watch(dioClientProvider));
});
```

import 'package:flutter_riverpod/flutter_riverpod.dart';

import '../../data/datasources/invitation_datasource.dart';
import '../../data/models/my_invitation_model.dart';

class MyInvitationsState {
  final List<MyInvitationModel> invitations;
  final bool isLoading;
  final Object? error;

  const MyInvitationsState({
    this.invitations = const [],
    this.isLoading = false,
    this.error,
  });

  MyInvitationsState copyWith({
    List<MyInvitationModel>? invitations,
    bool? isLoading,
    Object? error,
  }) =>
      MyInvitationsState(
        invitations: invitations ?? this.invitations,
        isLoading:   isLoading ?? this.isLoading,
        error:       error,
      );
}

class MyInvitationsNotifier extends StateNotifier<MyInvitationsState> {
  final InvitationDatasource _ds;
  MyInvitationsNotifier(this._ds) : super(const MyInvitationsState());

  Future<void> load() async {
    if (!mounted) return;
    state = state.copyWith(isLoading: true, error: null);
    try {
      final list = await _ds.listMine();
      if (!mounted) return;
      state = state.copyWith(invitations: list, isLoading: false);
    } catch (e) {
      if (!mounted) return;
      state = state.copyWith(isLoading: false, error: e);
    }
  }

  Future<void> accept(int id) async {
    final prev = state.invitations;
    state = state.copyWith(invitations: prev.where((i) => i.id != id).toList());
    try {
      await _ds.accept(id);
    } catch (_) {
      if (!mounted) return;
      state = state.copyWith(invitations: prev);
      rethrow;
    }
  }

  Future<void> refuse(int id) async {
    final prev = state.invitations;
    state = state.copyWith(invitations: prev.where((i) => i.id != id).toList());
    try {
      await _ds.refuse(id);
    } catch (_) {
      if (!mounted) return;
      state = state.copyWith(invitations: prev);
      rethrow;
    }
  }
}

final myInvitationsProvider =
    StateNotifierProvider<MyInvitationsNotifier, MyInvitationsState>((ref) {
  return MyInvitationsNotifier(ref.watch(invitationDatasourceProvider));
});
```

import 'package:app_links/app_links.dart';
import 'package:flutter/widgets.dart';
import 'package:go_router/go_router.dart';

class DeepLinkService {
  final AppLinks _appLinks = AppLinks();

  Future<void> init(GlobalKey<NavigatorState> navKey) async {
    final initial = await _appLinks.getInitialLink();
    if (initial != null) _handle(initial, navKey);

    _appLinks.uriLinkStream.listen((uri) => _handle(uri, navKey));
  }

  void _handle(Uri uri, GlobalKey<NavigatorState> navKey) {
    if (uri.scheme == 'terminim' && uri.host == 'invite') {
      final token = uri.pathSegments.isNotEmpty ? uri.pathSegments.first : null;
      if (token != null && token.length == 64) {
        navKey.currentContext?.go('/invite/$token');
      }
    } else if (uri.host.endsWith('terminim.com') && uri.pathSegments.isNotEmpty
        && uri.pathSegments.first == 'invite') {
      final token = uri.pathSegments.length >= 2 ? uri.pathSegments[1] : null;
      if (token != null && token.length == 64) {
        navKey.currentContext?.go('/invite/$token');
      }
    }
  }
}

import 'package:flutter_riverpod/flutter_riverpod.dart';

import '../../../../core/theme/app_colors.dart';
import '../../../../core/theme/app_spacing.dart';

import '../../../employee_invitation/data/models/owner_invitation_model.dart';

class PendingInvitationCard extends ConsumerWidget {
  final OwnerInvitationModel invitation;
  final VoidCallback onChanged;
  const PendingInvitationCard({super.key, required this.invitation, required this.onChanged});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final l = context.l10n;
    final ds = ref.watch(invitationDatasourceProvider);

    final initials = _initials(invitation);
    final remaining = invitation.expiresAt.difference(DateTime.now());

    return Container(
      decoration: BoxDecoration(
        color: AppColors.surface,
        borderRadius: BorderRadius.circular(AppSpacing.radiusLg),
        border: Border.all(color: AppColors.secondary.withValues(alpha: 0.4)),
      ),
      padding: const EdgeInsets.all(AppSpacing.md),
      child: Row(
        children: [
          CircleAvatar(
            backgroundColor: AppColors.ivoryAlt,
            child: Text(initials, style: AppTextStyles.body),
          ),
          const SizedBox(width: AppSpacing.sm),
          Expanded(
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                Text(_displayName(invitation), style: AppTextStyles.body),
                const SizedBox(height: 2),
                Row(
                  children: [
                    Container(
                      padding: const EdgeInsets.symmetric(horizontal: AppSpacing.sm, vertical: 2),
                      decoration: BoxDecoration(
                        color: AppColors.secondary.withValues(alpha: 0.15),
                        borderRadius: BorderRadius.circular(AppSpacing.radiusSm),
                      ),
                      child: Text(
                        l.invitationStatusPending,
                        style: AppTextStyles.overline
                            .copyWith(color: AppColors.secondaryDark),
                      ),
                    ),
                    const SizedBox(width: AppSpacing.sm),
                    Text(
                      l.invitationExpiresIn(duration: _formatRemaining(context, remaining)),
                      style: AppTextStyles.caption,
                    ),
                  ],
                ),
              ],
            ),
          ),
          PopupMenuButton<String>(
            onSelected: (value) async {
              if (value == 'resend') {
                await ds.resendInvitation(invitation.id);
              } else if (value == 'cancel') {
                await ds.revokeInvitation(invitation.id);
              }
              onChanged();
            },
            itemBuilder: (_) => [
              PopupMenuItem(value: 'resend', child: Text(l.invitationActionResend)),
              PopupMenuItem(value: 'cancel', child: Text(l.invitationActionCancel)),
            ],
          ),
        ],
      ),
    );
  }

  String _initials(OwnerInvitationModel inv) {
    if ((inv.firstName ?? '').isEmpty && (inv.lastName ?? '').isEmpty) return '@';
    final f = (inv.firstName ?? '').isNotEmpty ? inv.firstName![0] : '';
    final l = (inv.lastName ?? '').isNotEmpty ? inv.lastName![0] : '';
    return (f + l).toUpperCase();
  }

  String _displayName(OwnerInvitationModel inv) {
    final n = [inv.firstName ?? '', inv.lastName ?? '']
        .where((s) => s.isNotEmpty).join(' ');
    return n.isEmpty ? inv.email : n;
  }

  String _formatRemaining(BuildContext context, Duration d) {
    if (d.isNegative) return '0h';
    final days = d.inDays;
    final hours = d.inHours % 24;
    if (days > 0) return '${days}j ${hours}h';
    return '${hours}h';
  }
}

        SnackBar(content: Text(context.errorMessage(e))),
      );
    } finally {
      if (mounted) setState(() => _busy = false);
    }
